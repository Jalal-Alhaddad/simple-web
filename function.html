1. Function Declaration using function keyword:


function greet(name) {
  return "Hello, " + name + "!";
}

console.log(greet("Alice")); // Output: Hello, Alice!

2. Function Expression:


const greet = function(name) {
  return "Hello, " + name + "!";
};

console.log(greet("Bob")); // Output: Hello, Bob!

3. Arrow Function:


const greet = (name) => {
  return "Hello, " + name + "!";
};

console.log(greet("Charlie")); // Output: Hello, Charlie!

4. Arrow Function with Implicit Return:


const greet = (name) => "Hello, " + name + "!";

console.log(greet("David")); // Output: Hello, David!

5. Arrow Function with Single Parameter:


const greet = name => "Hello, " + name + "!";

console.log(greet("Emily")); // Output: Hello, Emily!

6. Arrow Function with No Parameters:


const greet = () => "Hello, World!";

console.log(greet()); // Output: Hello, World!

7. Arrow Function with Return Statement:


const add = (a, b) => {
  return a + b;
};

console.log(add(3, 5)); // Output: 8
