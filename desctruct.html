1. Array Destructuring:


const numbers = [1, 2, 3, 4, 5];

// Destructuring assignment
const [first, second, ...rest] = numbers;

console.log(first); // Output: 1
console.log(second); // Output: 2
console.log(rest); // Output: [3, 4, 5]

2. Object Destructuring:


const person = {
  name: 'Alice',
  age: 30,
  country: 'USA'
};

// Destructuring assignment
const { name, age, country } = person;

console.log(name); // Output: Alice
console.log(age); // Output: 30
console.log(country); // Output: USA

3. Function Parameter Destructuring:


// Function with object parameter
const printPersonInfo = ({ name, age, country }) => {
  console.log(`Name: ${name}, Age: ${age}, Country: ${country}`);
};

const person = {
  name: 'Bob',
  age: 25,
  country: 'Canada'
};

// Passing object to function with destructuring
printPersonInfo(person);
// Output: Name: Bob, Age: 25, Country: Canada

4. Default Values in Destructuring:


const person = {
  name: 'Charlie',
  age: 35
};

// Destructuring assignment with default value
const { name, age, country = 'Unknown' } = person;

console.log(name); // Output: Charlie
console.log(age); // Output: 35
console.log(country); // Output: Unknown

5. Nested Destructuring:


const user = {
  id: 123,
  info: {
    name: 'David',
    email: 'david@example.com'
  }
};

// Nested destructuring assignment
const { id, info: { name, email } } = user;

console.log(id); // Output: 123
console.log(name); // Output: David
console.log(email); // Output: david@example.com
