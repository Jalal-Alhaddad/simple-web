Creating a React Native Component without JSX:


const React = require('react');
const { View, Text } = require('react-native');

// Non-JSX way to create a component
const MyComponentWithoutJSX = function(props) {
  return React.createElement(
    View,
    null,
    React.createElement(
      Text,
      null,
      'Hello, ',
      props.name
    )
  );
};

module.exports = MyComponentWithoutJSX;

Creating a React Native Component with JSX:


import React from 'react';
import { View, Text } from 'react-native';

// JSX way to create a component
const MyComponentWithJSX = (props) => {
  return (
    
      Hello, {props.name}
    
  );
};

export default MyComponentWithJSX;
